52 bool ECUStates::PreCharge_State::voltageCheck() {
53  Canbus::setSemaphore(ADD_BMS_VOLT);
54  uint8_t BMSVolt = *((uint8_t *)(BMS_VOLT_Buffer + 4)); // Byte 5: voltage
55  Canbus::setSemaphore(ADD_MC0_VOLT);
56  int16_t MC0Volt = *((int16_t *)(MC0_VOLT_Buffer)) / 10; // Bytes 0-1: DC BUS MC Voltage
57  Canbus::setSemaphore(ADD_MC1_VOLT);
58  int16_t MC1Volt = *((int16_t *)(MC1_VOLT_Buffer)) / 10; // Bytes 0-1: DC BUS MC Voltage
59  Canbus::clearSemaphore();
60  return 0.9 * BMSVolt <= (MC0Volt + MC1Volt) / 2;
61 }

218  Log.i(ID, "Loading Buffers");
219  getBuffers();
220  Log.i(ID, "Driving mode on");
221 
222 #if CONF_ECU_POSITION == FRONT_ECU
223  elapsedMillis timeElapsed;
224 #endif
225 
226  while (true) {
227  Canbus::update();
228 #if CONF_ECU_POSITION == FRONT_ECU
229  if (timeElapsed >= 5) { // Update Tablet every 5ms
230  timeElapsed = 0;
231  // receive rpm of MCs, interpret, then send to from teensy for logging
232  Canbus::setSemaphore(ADD_MC0_RPM);
233  int16_t MC_Rpm_Val_0 = *(int16_t *)(MC0_RPM_Buffer + 2); // Bytes 2-3 : Angular Velocity
234  Canbus::setSemaphore(ADD_MC1_RPM);
235  int16_t MC_Rpm_Val_1 = *(int16_t *)(MC1_RPM_Buffer + 2); // Bytes 2-3 : Angular Velocity
236  Canbus::clearSemaphore();
237  float MC_Spd_Val_0 = wheelRadius * 2 * 3.1415926536 / 60 * MC_Rpm_Val_0;
238  float MC_Spd_Val_1 = wheelRadius * 2 * 3.1415926536 / 60 * MC_Rpm_Val_1;
239  float speed = (MC_Spd_Val_0 + MC_Spd_Val_1) / 2;
240  Log.i(ID, "Current Motor Speed:", speed);
241  Log.i(ID, "Current Power Value:", powerValue()); // Canbus message from MCs
242  Log.i(ID, "BMS State Of Charge Value:", BMSSOC()); // Canbus message
243  }
244 #else
245  Log.d(ID, "Sending Fault reset to MCs complete");
246 
247  for (size_t i = 0; i < 4; i++) { // IMPROVE: Send only once? Check MC heartbeat catches it
248  Canbus::sendData(ADD_MC0_CLEAR, 20, 0, 1); // NOTE: We are assuming MCs are little endian
249  Canbus::sendData(ADD_MC1_CLEAR, 20, 0, 1);
250  delay(10);
251  }
252 
253  int MotorTorques[2] = {0};
254  torqueVector(MotorTorques);
255 
256  // TODO: is the clear faults address the same as sending torque address?
257  sendMCCommand(ADD_MC0_CLEAR, MotorTorques[0], 0, 0); // MC 1
258  sendMCCommand(ADD_MC1_CLEAR, MotorTorques[1], 1, 0); // MC 2
259 
260  if (Pins::getCanPinValue(PINS_FRONT_BUTTON_INPUT)) {
261  Log.w(ID, "Going back to Idle state");
262  break;
263  }
264 
265  if (FaultCheck()) {
266  // TODO: anything extra when a fault happens while driving?
267  return &ECUStates::FaultState;
268  }
269 #endif
270  }
271 
272  Log.i(ID, "Driving mode off");
273  return &ECUStates::Idle_State;
